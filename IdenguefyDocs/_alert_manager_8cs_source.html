153 public void EvaluateAlerts()
154 {
155 Debug.Log($"[AlertManager] Evaluating alerts...");
156
157 // Check dependencies
158 if (clusterController == null || clusterController.Clusters.Count == 0)
159 {
160 Debug.LogWarning("[AlertManager] No clusters available yet.");
161 return;
162 }
163
164 if (locationController == null || !locationController.IsReady)
165 {
166 Debug.LogWarning("[AlertManager] LocationController not ready.");
167 return;
168 }
169
170 // Load pointers
171 PointerController pointerController = FindAnyObjectByType<PointerController>();
172 if (pointerController != null)
173 Debug.Log("[AlertManager] PointerController found dynamically.");
174 else
175 Debug.LogWarning("[AlertManager] PointerController not found at Start!");
176
177 List<MapPointer> allPointers = new List<MapPointer>();
178 if (pointerController != null)
179 {
180 allPointers = pointerController.ListPointers();
181 Debug.Log($"[AlertManager] Loaded {allPointers.Count} pointers for alert evaluation.");
182 }
183 else
184 {
185 Debug.LogError($"[PointerController] Missing!");
186 }
187
188 (float userLon, float userLat) = locationController.CurrentCoordinates;
189 int threshold = new SettingsController().GetThresholdPref();
190 Debug.LogWarning($"Threshold = {threshold}");
191 double thresholdKm = threshold / 1000.0;
192
193 // Evaluate all pointers
194 foreach (var pointer in allPointers)
195 {
196 foreach (var cluster in clusterController.Clusters)
197 {
198 double distance = MapUtils.HaversineDistance(
199 pointer.Coordinates.Item1, pointer.Coordinates.Item2,
200 cluster.Coordinates[0].lon, cluster.Coordinates[0].lat);
201
202 if (distance < thresholdKm)
203 {
204 DateTime now = DateTime.Now;
205 string key = pointer.Name;
206
207 if (!lastNotifyTimes.TryGetValue(key, out DateTime lastTime))
208 lastTime = DateTime.MinValue;
209
210 if ((now - lastTime).TotalSeconds >= durationThreshold)
211 {
212 lastNotifyTimes[key] = now;
213
214 AlertType type = pointer.HomeTag ? AlertType.Indoor : AlertType.Outdoor;
215
216 AlertData data = new AlertData
217 {
218 Title = pointer.HomeTag ? "Home Alert!" : "Pointer Alert!",
219 Message = $"{cluster.AreaName} cluster detected within {threshold}m of pointer '{pointer.Name}'",
220 Timestamp = now,
221 Type = type
222 };
223
224 Debug.Log($"[Alert Manager] Triggering alert for {pointer.Name}");
225 NotifySubscribers(data);
226 }
227 }
228 }
229 }
230
231 // Evaluate user's live location (Outdoor only)
232 foreach (var cluster in clusterController.Clusters)
233 {
234 double distance = MapUtils.HaversineDistance(
235 userLon, userLat,
236 cluster.Coordinates[0].lon, cluster.Coordinates[0].lat);
237
238 Debug.Log($"LIVE LOC coords: {userLon}, {userLat} compare with " +
239 $"cluster coords: {cluster.Coordinates[0].lon}, {cluster.Coordinates[0].lat}");
240
241 Debug.LogError($"DISTANCE LIVELOC ARGHHHHHHH {distance}, THRESHOLDKM {thresholdKm}");
242
243 if (distance < thresholdKm)
244 {
245 DateTime now = DateTime.Now;
246 string key = "LIVE";
247
248 if (!lastNotifyTimes.TryGetValue(key, out DateTime lastTime))
249 lastTime = DateTime.MinValue;
250
251 if ((now - lastTime).TotalSeconds >= durationThreshold)
252 {
253 lastNotifyTimes[key] = now;
254
255 AlertData data = new AlertData
256 {
257 Title = "Nearby Dengue Cluster!",
258 Message = $"{cluster.AreaName} cluster detected within {threshold}m of your current location",
259 Timestamp = now,
260 Type = AlertType.Outdoor
261 };
262
263 Debug.Log($"[Alert Manager] Triggering live location alert");
264 NotifySubscribers(data);
265 }
266 }
267 }
268
269 Debug.Log("[AlertManager] Evaluation complete.");
270 }
