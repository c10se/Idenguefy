104 private List<MapSearchResult> PreprocessSearchData(string json)
105 {
106 List<MapSearchResult> searchResultList = new List<MapSearchResult>();
107 JObject geoData = JObject.Parse(json);
108 JArray features = (JArray)geoData["features"];
109 foreach (var feature in features)
110 {
111 JObject geo = (JObject)feature["geometry"];
112 JArray coord = (JArray)geo["coordinates"];
113
114 float relevance = feature["relevance"].ToObject<float>();
115
116 MapSearchResult result = new MapSearchResult
117 (
118 name: feature["text"].ToString(),
119 detail: feature["place_name"].ToString(),
120 relevance: feature["relevance"].ToObject<float>(),
121 coord: (
122 coord[0].ToObject<float>(),
123 coord[1].ToObject<float>()
124 )
125
126 );
127 searchResultList.Add(result);
128 Debug.Log($"[MTSearchAPIHandler] name={result.name}, coord=({result.coord.lon}, {result.coord.lat})");
129 }
130 return searchResultList;
131 }
