158 public async Task<string> TranslateTextAsync(string sourceText, string targetLanguageCode)
159 {
160 if (string.IsNullOrWhiteSpace(sourceText))
161 return sourceText;
162
163 // Ignore text containing only punctuation or symbols
164 if (System.Text.RegularExpressions.Regex.IsMatch(sourceText, @"^[\p{P}\p{S}\s]+$"))
165 return sourceText;
166
167 sourceText = sourceText.Replace("\"", "\\\"");
168 string cacheKey = $"{sourceText}|{targetLanguageCode}";
169
170 if (translationCache.TryGetValue(cacheKey, out string cachedResult))
171 return cachedResult;
172
173 // Step 1: Temporarily replace punctuation with placeholders to preserve format
174 var punctuationRegex = new System.Text.RegularExpressions.Regex(@"[!?,.:;()/:]");
175 var punctuationMap = new Dictionary<string, string>();
176 int index = 0;
177
178 string textWithPlaceholders = punctuationRegex.Replace(sourceText, match =>
179 {
180 string token = $"<P{index++}>"; // e.g., <P0>, <P1>
181 punctuationMap[token] = match.Value;
182 return token;
183 });
184
185 // Step 2: Send the placeholder text to the Translator API
186 string endpoint = $"https://api.cognitive.microsofttranslator.com/translate?api-version=3.0&to={targetLanguageCode}";
187 string bodyJson = "[{\"Text\": \"" + textWithPlaceholders + "\"}]";
188
189 int retryCount = 0;
190 while (retryCount < MaxRetries)
191 {
192 using (UnityWebRequest request = new UnityWebRequest(endpoint, "POST"))
193 {
194 byte[] bodyRaw = System.Text.Encoding.UTF8.GetBytes(bodyJson);
195 request.uploadHandler = new UploadHandlerRaw(bodyRaw);
196 request.downloadHandler = new DownloadHandlerBuffer();
197 request.SetRequestHeader("Content-Type", "application/json");
198 request.SetRequestHeader("Ocp-Apim-Subscription-Key", API_KEY);
199 request.SetRequestHeader("Ocp-Apim-Subscription-Region", "southeastasia");
200
201 await request.SendWebRequest();
202
203 // Handle rate limiting with exponential backoff
204 if (request.responseCode == 429)
205 {
206 int waitTime = (int)Mathf.Pow(2, retryCount) * 1000;
207 Debug.LogWarning($"[LocalizationManager] 429 Too Many Requests. Retrying in {waitTime}ms...");
208 await Task.Delay(waitTime);
209 retryCount++;
210 continue;
211 }
212
213 // Handle other API errors
214 if (request.result != UnityWebRequest.Result.Success)
215 {
216 Debug.LogError($"[LocalizationManager] Translation API error: {request.error}");
217 return sourceText;
218 }
219
220 // Step 3: Parse and restore translated text
221 string json = request.downloadHandler.text;
222 try
223 {
224 string wrapped = "{\"items\":" + json + "}";
225 TranslationWrapper wrapper = JsonUtility.FromJson<TranslationWrapper>(wrapped);
226 if (wrapper.items.Length > 0 && wrapper.items[0].translations.Length > 0)
227 {
228 string translatedText = wrapper.items[0].translations[0].text;
229
230 // Restore punctuation
231 foreach (var kvp in punctuationMap)
232 translatedText = translatedText.Replace(kvp.Key, kvp.Value);
233
234 translationCache[cacheKey] = translatedText;
235 return translatedText;
236 }
237 }
238 catch
239 {
240 Debug.LogError($"[LocalizationManager] JSON parse failed: {json}");
241 }
242
243 break;
244 }
245 }
246
247 return sourceText;
248 }
