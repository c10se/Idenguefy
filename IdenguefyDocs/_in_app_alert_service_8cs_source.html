87 TMP_Text[] texts = alertObj.GetComponentsInChildren<TMP_Text>();
88
89 // Find and populate text fields by name
90 foreach (TMP_Text txt in texts)
91 {
92 string name = txt.name.ToLower();
93 if (name.Contains("title"))
94 txt.text = data.Title;
95 else if (name.Contains("message"))
96 txt.text = data.Message;
97 else if (name.Contains("timestamp"))
98 txt.text = data.Timestamp.ToString("dd MMM yyyy, HH:mm");
99 else if (txt.name.ToLower().Contains("type"))
100 txt.text = "Alert Type: " + data.Type.ToString();
101 }
102
103 // Hook up Show More button
104 Button showMoreButton = alertObj.GetComponentInChildren<Button>();
105 if (showMoreButton != null)
106 {
107 showMoreButton.onClick.RemoveAllListeners();
108 showMoreButton.onClick.AddListener(() =>
109 {
110 // Show the correct health tips based on the alert type
111 if (data.Type == AlertType.Indoor)
112 AnimationManager.Instance.ToggleIndoorHealthTips();
113 //healthTipsView.ShowIndoorTips();
114 else
115 AnimationManager.Instance.ToggleOutdoorHealthTips();
116 //healthTipsView.ShowOutdoorTips();
117 });
118 }
119
120 activeAlerts.Add(alertObj);
121 EnforceMaxVisibleAlerts();
122
123 Debug.Log($"[InAppAlertService] Displayed {data.Type} alert: {data.Title}");
124 }
