60 private void CreateCircle(DengueCluster cluster, int zoom, int xMin, int yMin, float tileSize)
61 {
62 //Find cluster centre
63 Vector2 centre = Vector2.zero;
64
65 foreach ((float lon, float lat) in cluster.Coordinates)
66 {
67 MapUtils.LonLatToPixelRatio(lon, lat, zoom, xMin, yMin,
68 out double xRatio, out double yRatio);
69
70 float xPixel = (float)(xRatio * tileSize);
71 float yPixel = (float)(yRatio * tileSize);
72
73 centre += new Vector2(xPixel, yPixel);
74 }
75 centre /= cluster.Coordinates.Count;
76
77 //Approximate radius
78 float radius = 0f;
79 foreach ((float lon, float lat) in cluster.Coordinates)
80 {
81 MapUtils.LonLatToPixelRatio(lon, lat, zoom, xMin, yMin,
82 out double xRatio, out double yRatio);
83
84 float xPixel = (float)(xRatio * tileSize);
85 float yPixel = (float)(yRatio * tileSize);
86
87 float dist = Vector2.Distance(centre, new Vector2(xPixel, yPixel));
88 if (dist > radius) radius = dist;
89 }
90
91 //Adjust for mapPanel pivot
92 // If pivot is center (0.5, 0.5) [WHICH IT IS, SO DO NOT CHANGE FROM 0.5, 0.5], move cluster coords so they align
93 Vector2 offset = new Vector2(mapPanelRect.sizeDelta.x / 2f, -mapPanelRect.sizeDelta.y / 2f);
94 Vector2 anchoredCentre = centre - offset;
95
96 //Create circle UI with a button
97 GameObject circleObj = new GameObject($"ClusterCircle_{cluster.LocationID}", typeof(Image), typeof(Image), typeof(Button));
98 circleObj.transform.SetParent(mapPanelRect, false);
99
100 Image img = circleObj.GetComponent<Image>();
101 img.sprite = circleSprite;
102 img.color = GetSeverityColor(cluster.Severity);
103
104 RectTransform rect = circleObj.GetComponent<RectTransform>();
105 rect.anchoredPosition = anchoredCentre;
106 rect.sizeDelta = new Vector2(radius * 2, radius * 2);
107
108
109 Button button = circleObj.GetComponent<Button>();
110 button.transition = Selectable.Transition.None;
111 //Dynamically create a listner, it's the "OnClick" in the inspector, but cannot see since it's dyanmically created
112 button.onClick.AddListener(() => OnClusterClicked(cluster, circleObj));
113
114 Debug.Log($"Cluster {cluster.AreaName} placed at {rect.anchoredPosition}, raw {centre}, offset {offset}"); //Debug to check offset
115 }
