63 private void RefreshClusterList()
64 {
65 // Clear old entries
66 foreach (var obj in clusterEntries)
67 Destroy(obj);
68 clusterEntries.Clear();
69
70 if (clusterController == null)
71 {
72 Debug.LogWarning("[ClusterView] ClusterController not assigned!");
73 return;
74 }
75
76 List<DengueCluster> clusters = clusterController.Clusters;
77
78 // Handle empty state
79 if (clusters == null || clusters.Count == 0)
80 {
81 if (emptyStateText != null)
82 emptyStateText.SetActive(true);
83 Debug.Log("[ClusterView] No clusters to display.");
84 return;
85 }
86
87 if (emptyStateText != null)
88 emptyStateText.SetActive(false);
89
90 // Populate list with new entries
91 foreach (DengueCluster cluster in clusters)
92 {
93 GameObject entry = Instantiate(clusterEntryPrefab, clusterEntryContainer);
94 TMP_Text[] texts = entry.GetComponentsInChildren<TMP_Text>();
95
96 // Find and populate text fields by name
97 foreach (TMP_Text txt in texts)
98 {
99 string lower = txt.name.ToLower();
100 if (lower.Contains("name"))
101 {
102 txt.text = $"<b>{cluster.AreaName}</b>\n";
103 }
104 if (lower.Contains("details"))
105 {
106 txt.text =
107 $"Cases: {cluster.CaseSize}\n" +
108 $"Severity: {cluster.Severity}\n" +
109 $"Location ID: {cluster.LocationID}";
110 }
111 }
112
113 clusterEntries.Add(entry);
114 }
115
116 Debug.Log($"[ClusterView] Displayed {clusters.Count} clusters.");
