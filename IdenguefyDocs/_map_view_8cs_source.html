199 public void ReadTouch()
200 {
201 if (!enableReadTouch)
202 {
203 return; // Do nothing if touch reading is disabled
204 }
205
206 if (Touchscreen.current != null)
207 {
208 foreach (var t in UnityEngine.InputSystem.EnhancedTouch.Touch.activeTouches)
209 {
210 //Debug.Log($"Value of touch phase upon touch: {t.phase}");
211 switch (t.phase)
212 {
213 case UnityEngine.InputSystem.TouchPhase.Began:
214 isHolding = false;
215 break;
216 case UnityEngine.InputSystem.TouchPhase.Moved:
217 case UnityEngine.InputSystem.TouchPhase.Stationary:
218 double holdTimer = t.time - t.startTime;
219 if (holdTimer >= longPressTime && !isHolding)
220 {
221 // Long press detected
222 isHolding = true;
223 TouchCoord = t.screenPosition;
224 //Coord = (TouchCoord.x, TouchCoord.y); // Unused
225 pointerView.OnLongPressDetected(TouchCoord);
226 }
227 break;
228 case UnityEngine.InputSystem.TouchPhase.Ended:
229 case UnityEngine.InputSystem.TouchPhase.Canceled:
230 // Reset hold state
231 isHolding = false;
232 holdTimer = 0;
233 break;
234 }
235 }
236 }
237 }
