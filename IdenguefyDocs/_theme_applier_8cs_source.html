94 public void ApplyTheme(ThemeData theme)
95 {
96 if (theme == null) return;
97
98 // --- IMAGES --- (Applying background color)
99 foreach (var img in imageTargets)
100 if (img != null) img.color = theme.backgroundColor;
101
102 // --- TEXTS ---
103 foreach (var txt in textTargets)
104 if (txt != null) txt.color = theme.textColor;
105
106 // --- BUTTONS ---
107 foreach (var btn in buttonTargets)
108 {
109 if (btn == null) continue;
110
111 var btnImage = btn.GetComponent<Image>();
112 if (btnImage != null)
113 btnImage.color = theme.buttonColor;
114
115 TMP_Text btnText = btn.GetComponentInChildren<TMP_Text>();
116 if (btnText != null)
117 btnText.color = theme.buttonTextColor;
118 }
119
120 // --- INPUT FIELDS ---
121 foreach (var field in inputFieldTargets)
122 {
123 if (field == null) continue;
124
125 var bg = field.GetComponent<Image>();
126 if (bg != null)
127 bg.color = theme.inputBackgroundColor;
128
129 if (field.textComponent != null)
130 field.textComponent.color = theme.inputTextColor;
131
132 if (field.placeholder is TMP_Text placeholder)
133 placeholder.color = theme.placeholderTextColor;
134 }
135
136 // --- TOGGLES ---
137 foreach (var toggle in toggleTargets)
138 {
139 if (toggle == null) continue;
140
141 if (toggle.targetGraphic is Image bg)
142 bg.color = toggle.isOn ? theme.toggleOnColor : theme.toggleOffColor;
143
144 if (toggle.graphic is Image check)
145 check.color = theme.textColor;
146
147 TMP_Text label = toggle.GetComponentInChildren<TMP_Text>();
148 if (label != null)
149 label.color = theme.textColor;
150 }
151
152 // --- SLIDERS ---
153 foreach (var slider in sliderTargets)
154 {
155 if (slider == null) continue;
156
157 if (slider.fillRect != null)
158 {
159 var fillImg = slider.fillRect.GetComponent<Image>();
160 if (fillImg != null)
161 fillImg.color = theme.sliderFillColor;
162 }
163
164 var bg = slider.GetComponent<Image>();
165 if (bg != null)
166 bg.color = theme.sliderBackgroundColor;
167
168 if (slider.handleRect != null)
169 {
170 var handleImg = slider.handleRect.GetComponent<Image>();
171 if (handleImg != null)
172 handleImg.color = theme.sliderFillColor;
173 }
174 }
175
176 // --- DROPDOWNS ---
177 foreach (var dropdown in dropdownTargets)
178 {
179 if (dropdown == null) continue;
180
181 // Background
182 var bg = dropdown.GetComponent<Image>();
183 if (bg != null)
184 bg.color = theme.dropdownBackgroundColor;
185
186 // Caption Text
187 if (dropdown.captionText != null)
188 dropdown.captionText.color = theme.dropdownTextColor;
189
190 // Arrow Icon (if exists)
191 var arrow = dropdown.transform.Find("Arrow")?.GetComponent<Image>();
192 if (arrow != null)
193 arrow.color = theme.dropdownArrowColor;
194
195 // Highlight color in dropdown template (if open)
196 if (dropdown.template != null)
197 {
198 // This logic might need refinement to correctly target the highlight graphic
199 var itemBg = dropdown.template.GetComponentInChildren<Image>();
200 if (itemBg != null)
201 itemBg.color = theme.dropdownHighlightColor;
202 }
203 }

256 private void ApplyTheme(ThemeMode theme)
257 {
258 int changed = 0;
259
260 // Update all UI Images (includes icons inside buttons, toggles, etc.)
261 foreach (var img in allImages)
262 {
263 if (img == null || img.sprite == null) continue;
264 Sprite swapped = GetSwappedSprite(img.sprite, theme);
265 if (swapped != null)
266 {
267 img.sprite = swapped;
268 changed++;
269 }
270 }
271
272 // Update all SpriteRenderers (for in-world sprites)
273 foreach (var sr in allSpriteRenderers)
274 {
275 if (sr == null || sr.sprite == null) continue;
276 Sprite swapped = GetSwappedSprite(sr.sprite, theme);
277 if (swapped != null)
278 {
279 sr.sprite = swapped;
280 changed++;
281 }
282 }
283
284 Debug.Log($"[ThemeSwapper] Applied {theme} theme {changed} sprites swapped.");
285 }

291 private void CacheAllSprites()
292 {
293 spriteCache.Clear();

295 // Load all sprites from Resources (so we can swap at runtime)
296 Sprite[] allSprites = Resources.LoadAll<Sprite>("");
297
298 foreach (var s in allSprites)

300 if (!spriteCache.ContainsKey(s.name))
301 spriteCache.Add(s.name, s);

314 private Sprite GetSwappedSprite(Sprite currentSprite, ThemeMode targetTheme)
315 {
316 if (currentSprite == null) return null;
317
318 string currentName = currentSprite.name;
319 string targetName;
320
321 // Naming convention:
322 if (targetTheme == ThemeMode.Dark)
323 targetName = currentName.Replace("Light Mode", "Dark Mode");
324 else
325 targetName = currentName.Replace("Dark Mode", "Light Mode");
326
327 if (targetName == currentName)
328 return null; // Already the correct sprite or not part of the theme system
329
330 // Try cache first
331 if (spriteCache.TryGetValue(targetName, out Sprite swapped))
332 return swapped;
333
334 // Fallback to direct Resources load (slower, but a good safety net)
335 swapped = Resources.Load<Sprite>(targetName);
336 if (swapped == null)
337 {
338 // Only log this warning if the sprite *should* have been swapped
339 Debug.LogWarning($"[ThemeSwapper] Missing sprite: '{targetName}' (from '{currentName}')");
340 }
341
342 return swapped;
343 }
