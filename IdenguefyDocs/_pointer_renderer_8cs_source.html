113 private void CreatePointerMarker(MapPointer pointer, int zoom, int xMin, int yMin, float tileSize)
114 {
115 //Find cluster centre
116 Vector2 centre = Vector2.zero;
117
118 float lon = pointer.Longitude;
119 float lat = pointer.Latitude;
120
121 // Convert geo-coordinates to pixel coordinates
122 MapUtils.LonLatToPixelRatio(lon, lat, zoom, xMin, yMin,
123 out double xRatio, out double yRatio);
124
125 float xPixel = (float)(xRatio * tileSize);
126 float yPixel = (float)(yRatio * tileSize);
127
128 centre = new Vector2(xPixel, yPixel);
129
130 //Adjust for mapPanel pivot (0.5, 0.5)
131 Vector2 offset = new Vector2(mapPanelRect.sizeDelta.x / 2f, -mapPanelRect.sizeDelta.y / 2f);
132 Vector2 anchoredCentre = centre - offset;
133
134 //Create pointer UI with a button
135 GameObject pointerObj = new GameObject($"Pointer_{pointer.MapID}", typeof(Image), typeof(Button));
136 pointerObj.transform.SetParent(mapPanelRect, false);
137
138 // Set the correct sprite based on theme and HomeTag
139 Image img = pointerObj.GetComponent<Image>();
140 bool darkMode = new SettingsController().GetThemePref();
141 if (darkMode)
142 {
143 if (pointer.HomeTag == true) { img.sprite = pointerDarkHomeSprite; }
144 else { img.sprite = pointerDarkSprite; }
145 }
146 else
147 {
148 if (pointer.HomeTag == true) { img.sprite = pointerHomeSprite; }
149 else { img.sprite = pointerSprite; }
150 }
151 img.color = new Color(img.color.r, img.color.g, img.color.b, 0.8f);
152
153 // Set size and position
154 RectTransform rect = pointerObj.GetComponent<RectTransform>();
155 rect.sizeDelta = new Vector2(100, 150);
156 rect.anchoredPosition = anchoredCentre;
157
158 // Configure the button component
159 Button button = pointerObj.GetComponent<Button>();
160 button.transition = Selectable.Transition.None; // No visual transition
161 //Dynamically create a listner, it's the "OnClick" in the inspector, but cannot see since it's dyanmically created
162 button.onClick.AddListener(() => OnPointerClicked(pointer, pointerObj));
163 //pointersObjList.Add(pointerObj);
164 pointerObjDict.Add(pointerObj, pointer);
165 Debug.Log($"[PointerRenderer] CreatePointerMarker() Pointer: {pointer.Name}, placed at {rect.anchoredPosition}, located at:{pointer.Coordinates} , raw {centre}, offset {offset}"); //Debug to check offset
166 }
