99 txt.text = alert.Timestamp.ToString("dd MMM yyyy, HH:mm");
100 else if (txt.name.ToLower().Contains("type"))
101 txt.text = "Alert Type: " + alert.Type.ToString();
102 }

104 activeAlertItems.Add(item);
