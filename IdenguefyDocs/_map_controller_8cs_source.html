94 public IEnumerator LoadMapTile(int x, int y, int xMin, int yMin)
95 {
96 Texture2D tileTexture = new Texture2D(2, 2);
97 Task<Texture2D> task = tileCacheManager.RetrieveCachedTile(x, y);
98 yield return new WaitUntil(() => task.IsCompleted);
99
100 // Load from cache if available
101 if (task.Result != null)
102 {
103 tileTexture = task.Result;
104 Debug.Log($"[MapController] Loaded tile ({x}, {y}) from cache.");
105 }
106 else
107 {
108 // Otherwise fetch from the API
109 Debug.Log("[MapController] Cache miss, fetching from API...");
110 yield return tileAPIHandler.FetchMapTile(x, y, ZOOM);
111 tileTexture = TileAPIHandler.response.texture;
112
113 if (tileTexture && !tileCacheManager.checkAnyPendingLoad(x, y))
114 tileCacheManager.CacheTileData(tileTexture, x, y);
115
116 Debug.Log($"[MapController] Loaded tile ({x}, {y}) from API.");
117 }
118
119 // Construct and render the tile in the map panel
120 GameObject mapTile = ConstructMapTile(x, y, xMin, yMin);
121 RawImage tileRawImage = mapTile.GetComponent<RawImage>();
122 tileRawImage.texture = tileTexture;
123 }

173 public (float lon, float lat) ScreenToGeo(Vector2 screenPos)
174 {
175 // Convert from screen space to local map coordinates
176 RectTransformUtility.ScreenPointToLocalPointInRectangle(
177 mapPanelRect,
178 screenPos,
179 null, // For Screen Space - Overlay canvas; otherwise use the camera reference
180 out Vector2 local
181 );
182
183 // Normalize local coordinates to a [0,1] range
184 Vector2 size = mapPanelRect.rect.size;
185 float x01 = Mathf.Clamp01((local.x + size.x * 0.5f) / size.x);
186 float y01 = Mathf.Clamp01((local.y + size.y * 0.5f) / size.y);
187
188 // Map normalized coordinates into Singapore’s longitude/latitude bounds
189 float lon = Mathf.Lerp(BOUNDARY[0, 0], BOUNDARY[1, 0], x01);
190 float lat = Mathf.Lerp(BOUNDARY[0, 1], BOUNDARY[1, 1], y01);
191
192 return (lon, lat);
193 }
