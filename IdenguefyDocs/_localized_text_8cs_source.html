91 public async Task UpdateTextsAsync(string targetLanguage)
92 {
93 if (LocalizationManager.Instance == null) return;
94
95 string apiCode = LocalizationManager.Instance.GetAPICode(targetLanguage);
96
97 foreach (var entry in localizedEntries)
98 {
99 if (entry.textComponent == null) continue;
100
101 // Store the original text if it hasn't been stored yet
102 if (string.IsNullOrEmpty(entry.originalText))
103 entry.originalText = entry.textComponent.text;
104
105 if (targetLanguage == "English")
106 {
107 // Revert to original text
108 entry.textComponent.text = entry.originalText;
109 }
110 else
111 {
112 // Translate from the original text
113 string translated = await LocalizationManager.Instance.TranslateTextAsync(entry.originalText, apiCode);
114 entry.textComponent.text = translated;
115 }
116 }

124 [ContextMenu("Auto Populate Text Elements")]
125 private void AutoPopulateTextElements()
126 {

130 TMP_Text[] allTexts = FindObjectsByType<TMP_Text>(FindObjectsSortMode.None);
131
132 foreach (var txt in allTexts)
133 {
134 if (txt == null) continue;
135
136 // Skip if the text object or any of its parents has the "IgnoreLocalisation" tag
137 // or if the text itself contains an ignore flag.
138 if (HasIgnoreParent(txt.transform) || txt.text.Contains("<Ignore>"))
139 continue;
140
141 localizedEntries.Add(new LocalizedEntry
142 {
143 textComponent = txt,
144 originalText = txt.text
145 });
146 trackedTexts.Add(txt);
147 }
148
149 Debug.Log($"[LocalizedText] Auto-populated {localizedEntries.Count} text elements (excluding ignored parents).");
150 }

157 private System.Collections.IEnumerator WatchForNewTextElements()
158 {
159 while (isWatching)
160 {
161 foreach (var txt in FindObjectsByType<TMP_Text>(FindObjectsSortMode.None))
162 {
163 if (txt == null || trackedTexts.Contains(txt)) continue;
164
165 // Add the ignore check here too for runtime objects
166 if (HasIgnoreParent(txt.transform) || txt.text.Contains("<Ignore>"))
167 continue;
168
169 // Found a new text element
170 trackedTexts.Add(txt);
171 localizedEntries.Add(new LocalizedEntry { textComponent = txt, originalText = txt.text });
172
173 Debug.Log($"[LocalizedText] ðŸ†• Detected new text: {txt.text} on {txt.gameObject.name}");
174 // Immediately translate the new text to the current language
175 _ = ApplyCurrentLanguageToTextAsync(txt);
176 }
177
178 yield return new WaitForSeconds(scanInterval);
179 }
